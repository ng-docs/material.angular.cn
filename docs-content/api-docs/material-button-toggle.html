import {MatButtonToggleModule} from '@angular/material/button-toggle';

指令

MatButtonToggleGroup

排他性按钮开关组的行为类似于单选按钮组。

选择器： mat-button-toggle-group

Exported as: matButtonToggleGroup
属性
名称 描述
@Input()

appearance: MatButtonToggleAppearance

该组中所有按钮的外观。
@Input({ transform: booleanAttribute })

disabled: boolean

是否禁用了多按钮开关组。
@Input({ transform: booleanAttribute })

hideMultipleSelectionIndicator: boolean

是否隐藏多选按钮切换组的复选标记指示器。
@Input({ transform: booleanAttribute })

hideSingleSelectionIndicator: boolean

是否隐藏单选按钮切换组的复选标记指示器。
@Input({ transform: booleanAttribute })

multiple: boolean

是否可以选择多个按钮开关。
@Input()

name: string

底层 input 元素的 name 属性。
@Input()

value: any

开关组的值。
@Input({ transform: booleanAttribute })

vertical: boolean

开关组是否为垂直的。
@Output()

change: EventEmitter<MatButtonToggleChange>

当组的值发生变化时会发出本事件。

selected: MatButtonToggle | MatButtonToggle[]

本组中选定的按钮开关。

MatButtonToggle

开关组内部的单个按钮。

选择器： mat-button-toggle

Exported as: matButtonToggle
属性
名称 描述
@Input()

appearance: MatButtonToggleAppearance

此按钮的外观样式。
@Input('aria-label')

ariaLabel: string

附着在宿主元素的 aria-label 属性上。在大多数情况下，aria-labelledby 优先，所以这个可以省略。
@Input('aria-labelledby')

ariaLabelledby: string | null

用户可以指定 aria-labelledby 属性，它会被转发到 input 元素
@Input({ transform: booleanAttribute })

checked: boolean

此按钮是否被勾选。
@Input({ transform: booleanAttribute })

disableRipple: boolean

是否在按钮切换上禁用波纹。
@Input({ transform: booleanAttribute })

disabled: boolean

该按钮是否已禁用。
@Input()

id: string

此按钮开关的唯一 ID。
@Input()

name: string

HTML 的 'name' 属性用于对单选按钮进行分组以进行唯一选择。
@Input()

value: any

MatButtonToggleGroup 读取它来指定自己的值。
@Output()

change: EventEmitter<MatButtonToggleChange>

本组的值发生变化时发出的事件。

buttonId: string

button 元素的唯一 ID。

buttonToggleGroup: MatButtonToggleGroup

父开关组（排他选择）。可选。

接口

MatButtonToggleDefaultOptions

表示按钮开关的默认选项，可用 MAT_BUTTON_TOGGLE_DEFAULT_OPTIONS 注入令牌进行配置。

属性
名称 描述

appearance: MatButtonToggleAppearance

切换按钮使用的默认外观。可以通过在切换按钮或组上显式设置外观来覆盖。

hideMultipleSelectionIndicator: boolean

是否应隐藏多选按钮切换组的图标指示器。

hideSingleSelectionIndicator: boolean

是否应该为单选按钮切换组隐藏图标指示器。

函数

MAT_BUTTON_TOGGLE_GROUP_DEFAULT_OPTIONS_FACTORY
返回值
MatButtonToggleDefaultOptions

类型别名

ToggleType

Deprecated 
type ToggleType = 'checkbox' | 'radio';

MatButtonToggleAppearance

按钮开关可能的外观样式。

type MatButtonToggleAppearance = 'legacy' | 'standard';

常量

MAT_BUTTON_TOGGLE_DEFAULT_OPTIONS

注入令牌，可以用来配置应用中所有按钮开关的默认选项。

const MAT_BUTTON_TOGGLE_DEFAULT_OPTIONS: InjectionToken<MatButtonToggleDefaultOptions>;

MAT_BUTTON_TOGGLE_GROUP

这个注入令牌可以用来引用 MatButtonToggleGroup 实例。它可以作为实际 MatButtonToggleGroup 类的备用令牌，使用实际类会导致无法优化不必要的类及其组件元数据。

const MAT_BUTTON_TOGGLE_GROUP: InjectionToken<MatButtonToggleGroup>;
